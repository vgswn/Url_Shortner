def self.custom_search(params)
  field = params[:field]+".trigram"
  query = params[:query] 
  urls = self.__elasticsearch__.search(
  {
   query: {
    bool: {
     must: [{
      term: {
       "#{field}":"#{query}"
      }
     }]
    }
   }
  }).records
  return urls
end
