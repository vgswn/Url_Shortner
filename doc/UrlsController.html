def convert_long_url_to_short_url
  if params[:long_url] == ""
    flash[:Error] = "Please Enter all Details"
    redirect_to urls_long_url_to_short_url_path
  else
    params[:long_url] = params[:long_url].strip
    @result = shorten_url(url_params)
    render '/urls/show'
  end
end
